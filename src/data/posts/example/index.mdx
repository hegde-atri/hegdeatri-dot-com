export async function getStaticPaths() {
  let allPosts = await Astro.glob("../../data/posts/**/*.md*");
  const posts = allPosts.filter((post) => !post.frontmatter.draft);
  return posts.map((post) => {
    return {
      params: { slug: post.frontmatter.slug },
      props: { ...post },
    };
  });
}
